001    package org.junit.function;
002    
003    /**
004     * This interface facilitates the use of
005     * {@link org.junit.Assert#assertThrows(Class, ThrowingRunnable)} from Java 8. It allows method
006     * references to void methods (that declare checked exceptions) to be passed directly into
007     * {@code assertThrows}
008     * without wrapping. It is not meant to be implemented directly.
009     *
010     * @since 4.13
011     */
012    public interface ThrowingRunnable {
013        void run() throws Throwable;
014    }
